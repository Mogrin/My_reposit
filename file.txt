import java.util.*;
import java.lang.*;
import java.io.*;

class Ideone
{
	public static void main (String[] args) throws java.lang.Exception
	{
		double a, b, c;
    	a = 0;
    	b = 2;
    	while (b - a > 1e-10)
    	{
    		c = (a + b) / 2;
        if(f(c) >= 0)
            b = c;
        else
            a = c;
    	}
    	System.out.println((a + b) / 2);	
    }
	
	void f(double x)
	{
		return exp(x) - (x + 2);
	}
}
